// JARVIS MARK II - MAIN INTEGRATION CORE
const API_KEY = 'PASTE_YOUR_API_KEY_HERE'; 

const chatBox = document.getElementById('chatBox');
const userInput = document.getElementById('userInput');

// Local storage logic for persistent memory matrix
let brotherName = localStorage.getItem('brother_name') || '';

// Trigger message sequence on Enter key press
userInput.addEventListener('keypress', function(e) {
    if (e.key === 'Enter') sendMessage();
});

async function sendMessage() {
    const text = userInput.value.trim();
    if (!text) return;

    // Display user input on HUD console
    appendMessage(text, 'user-msg', 'You: ');
    userInput.value = '';

    const lowerText = text.toLowerCase();

    // 1. Local Database & Core Command Handling
    if (lowerText.includes('my brother name is') || lowerText.includes("my brother's name is")) {
        let name = text.split(/is/i)[1].trim();
        localStorage.setItem('brother_name', name);
        brotherName = name;
        let msg = `Memory sector updated. Registered brother's name as ${name}, boss.`;
        appendMessage(msg, 'jarvis-msg', 'Jarvis: ');
        speak(msg);
        return;
    }

    if (lowerText.includes('brother name') || lowerText.includes("brother's name") || lowerText.includes('who is my brother')) {
        let msg = brotherName ? `Accessing localized memory units. Your brother's name is ${brotherName}.` : "I do not have your brother's name registered in my local mainframe database yet, boss.";
        appendMessage(msg, 'jarvis-msg', 'Jarvis: ');
        speak(msg);
        return;
    }

    if (lowerText.includes('play') || lowerText.includes('open youtube')) {
        let song = lowerText.replace('open youtube play this song', '').replace('play', '').trim();
        speak("Opening YouTube, boss.");
        window.open(`https://www.youtube.com/results?search_query=${encodeURIComponent(song)}`, '_blank');
        appendMessage("Opening YouTube mainframe...", 'jarvis-msg', 'Jarvis: ');
        return;
    }

    if (lowerText.includes('open google')) {
        speak("Opening Google, boss.");
        window.open('https://www.google.com', '_blank');
        appendMessage("Redirecting to Google...", 'jarvis-msg', 'Jarvis: ');
        return;
    }

    // 2. Cloud AI Mainframe (Google Gemini API Interface)
    try {
        appendMessage("Thinking...", 'jarvis-msg', 'Jarvis: ');
        
        const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${API_KEY}`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                contents: [{
                    parts: [{
                        text: `You are Jarvis, the highly intelligent AI assistant to Mr. Shriniwas Mahadik (the boss). Keep responses exceptionally short, smart, and brief like Iron Man's AI. Understand English, Hindi, and Marathi perfectly. Respond in the exact same language the boss used: ${text}`
                    }]
                }]
            })
        });

        const data = await response.json();
        
        // Remove 'Thinking...' loading text block
        chatBox.removeChild(chatBox.lastChild); 
        
        if (data.candidates && data.candidates[0].content.parts[0].text) {
            const aiResponse = data.candidates[0].content.parts[0].text.trim();
            appendMessage(aiResponse, 'jarvis-msg', 'Jarvis: ');
            speak(aiResponse);
        } else {
            appendMessage("Mainframe returned an empty response, boss.", 'jarvis-msg', 'Jarvis: ');
        }

    } catch (error) {
        console.error(error);
        if (chatBox.lastChild && chatBox.lastChild.innerText.includes("Thinking...")) {
            chatBox.removeChild(chatBox.lastChild);
        }
        appendMessage("Mainframe link error. Please verify the API key integrity matrix.", 'jarvis-msg', 'Jarvis: ');
    }
}

function appendMessage(text, className, prefix) {
    const messageElement = document.createElement('div');
    messageElement.className = className;
    messageElement.innerText = prefix + text;
    chatBox.appendChild(messageElement);
    chatBox.scrollTop = chatBox.scrollHeight;
}

// System Speech Synthesis Matrix
function speak(text) {
    if ('speechSynthesis' in window) {
        window.speechSynthesis.cancel(); // Clears previous speech streams
        const utterance = new SpeechSynthesisUtterance(text);
        utterance.lang = 'en-US'; 
        window.speechSynthesis.speak(utterance);
    }
}
